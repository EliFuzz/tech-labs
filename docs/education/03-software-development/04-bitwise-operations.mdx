import Tabs from '@theme/Tabs';

<Tabs queryString="primary">
    <TabItem value="overview" label="Overview">
        **Binary operation**: manipulation of individual bits in binary data or numbers.
    </TabItem>
    <TabItem value="negative-number-conversion" label="Negative number conversion">
        - Convert number into 2's complement (binary): `7 -> 0111`
        - Invert all the bits: `0111 -> 1000`
        - Add 1: `1000 -> 1001`
    </TabItem>
    <TabItem value="bitwise-operations" label="Bitwise operations">
        <table>
          <thead>
            <tr>
              <th style={{minWidth: '220px'}}>Operator</th>
              <th>Description</th>
              <th>Example</th>
            </tr>
          </thead>
          <tbody>
            <tr>
              <td>AND (`&`)</td>
              <td>If both bits are `1`, the result is `1`; otherwise `0`</td>
              <td>
              ```text
              0101 (5) &
              0011 (3)
              ----
              0001 (1)
              ```
              </td>
            </tr>
            <tr>
              <td>OR (`|`)</td>
              <td>If any bit is `1`, the result is `1`; otherwise `0`</td>
              <td>
              ```text
              0101 (5) |
              0011 (3)
              ----
              0111 (7)
              ```
              </td>
            </tr>
            <tr>
              <td>XOR: exclusive OR (`^`)</td>
              <td>If the bits are different, the result is `1`; otherwise `0`</td>
              <td>
              ```text
              0101 (5) ^
              0011 (3)
              ----
              0110 (6)
              ```
              </td>
            </tr>
            <tr>
              <td>NOT (`~`)</td>
              <td>Flips the bits. Each `0` becomes `1`, and each `1` becomes `0`</td>
              <td>
              ```text
              0000 0101 (5) ~
              --------
              1111 1010 (-6)
              ```
              </td>
            </tr>
            <tr>
              <td>Signed Left Shift (`<<`)</td>
              <td>Shifts the bits of the left operand to the left by a number of positions specified by the right operand</td>
              <td>
              ```text
              # positive number

              0000 0101 (5) <<
              0000 0010 (2)
              ---------
              0001 0100 (20)

              # negative number

              1111 1011 (-5) <<
              0000 0010 (2)
              ---------
              1110 1100 (-20)
              ```
              </td>
            </tr>
            <tr>
              <td>Signed Right Shift (`>>`)</td>
              <td>Shifts the bits of the left operand to the right by a number of positions specified by the right operand</td>
              <td>
              ```text
              # positive number

              0101 (5) >>
              0010 (2)
              ----
              0001 (1)

              # negative number

              1111 1011 (-5) >>
              0000 0010 (2)
              ---------
              1111 1110 (-2)
              ```
              </td>
            </tr>
            <tr>
              <td>Unsigned Right Shift (`>>>`)</td>
              <td>Similar to the right shift operator, but it treats the value as an unsigned integer. It fills the vacant bits on the left with zeros</td>
              <td>
              ```text
              # positive number

              0101 (5) >>>
              0010 (2)
              ----
              0001 (1)

              # negative number

              1111 0101 (-5) >>>
              0000 0010 (2)
              ---------
              0111 1111 1111 1111 1111 1111 1111 1110 (1073741822)
              ```
              </td>
            </tr>
          </tbody>
        </table>
    </TabItem>
</Tabs>
